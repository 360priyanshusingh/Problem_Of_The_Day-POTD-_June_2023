int leastInterval(int N, int k, vector<char> &tasks) {
        // code here
        vector<int> v(26,0);
        for(auto x:tasks)++v[x-'A'];
        int n=*max_element(v.begin(),v.end());
        int f=0, res=(n-1)*k+n;
        for(int i=0;i<26;++i)
        {
            if(v[i]<n)continue;
            if(v[i]==n&&f==0)f=1;
            else if(v[i]==n)
            {
                ++res;
            }
        }
        return max(res,N);
    }
